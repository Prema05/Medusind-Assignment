--CREATING STORED PROCEDURE FOR INSERT DATA--

CREATE PROCEDURE InsertEmp
@EMPNO DECIMAL(4),
@ENAME VARCHAR(15),
@JOB VARCHAR(15),
@MGR DECIMAL(4),
@SAL DECIMAL(7,2),
@COMM DECIMAL(7,2),
@DEPTNO DECIMAL(7,2)

AS
BEGIN
INSERT INTO EMP(EMPNO, ENAME, JOB, MGR, SAL, COMM, DEPTNO)
VALUES(@EMPNO, @ENAME, @JOB, @MGR, @SAL, @COMM, @DEPTNO)
END

--CREATING STORED PROCEDURE FOR DELECTE DATA-- 

CREATE PROCEDURE DeleteEmp
@EMPNO DECIMAL(4)

AS
BEGIN
DELETE FROM EMP WHERE EMPNO = @EMPNO
END

--CREATING STORED PROCEDURE FOR UPDATE DATA--

CREATE PROCEDURE UpdateEmp
@EMPNO DECIMAL(4),
@ENAME VARCHAR(15),
@SAL DECIMAL(7,2)

AS
BEGIN
UPDATE EMP
SET ENAME = @ENAME, SAL = @SAL
WHERE EMPNO = @EMPNO
END

--CREATING STORED PROCEDURE FOR SELECT DATA--

CREATE PROCEDURE SelectEmp
AS
BEGIN
SELECT * FROM EMP
END
